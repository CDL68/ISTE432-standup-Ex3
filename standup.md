```QUERY1
SELECT LAN.LANGUAGE AS First_Language
FROM USERLANG ULAN
INNER JOIN LANGUAGE LAN
ON ULAN.LANGUAGE_ID = LAN.LANGUAGE_ID
GROUP BY ULAN.LANGUAGE_ID
ORDER BY COUNT(*) DESC
LIMIT 1;
```



```QUERY2
SELECT LAN.LANGUAGE AS Language, 
(case WHEN SKILL_LEVEL = 1 then 'HIGH' 
	WHEN SKILL_LEVEL=2 then 'MEDIUM' 
	WHEN SKILL_LEVEL=3 then'LOW' 
    WHEN SKILL_LEVEL=4 then'ULTRA LOW' 
	ELSE NULL END
) AS Skill_Level,
COUNT(
	case 
    when SKILL_LEVEL = 1 THEN 1 
    when SKILL_LEVEL = 2 THEN 1
    when SKILL_LEVEL = 3 THEN 1 
    when SKILL_LEVEL = 4 THEN 1 
    ELSE NULL END) AS COUNT
FROM USERLANG ULAN
INNER JOIN LANGUAGE LAN
ON ULAN.LANGUAGE_ID = LAN.LANGUAGE_ID
GROUP BY LAN.LANGUAGE,Skill_Level
ORDER BY LAN.LANGUAGE ASC;
```

```QUERY3-1
SELECT DISTINCT ULAN_MAIN.STUDENT_ID,
	(SELECT LAN.LANGUAGE FROM USERLANG ULAN
		INNER JOIN LANGUAGE LAN
		ON ULAN.LANGUAGE_ID = LAN.LANGUAGE_ID
		WHERE ULAN.STUDENT_ID = ULAN_MAIN.STUDENT_ID
		AND ULAN.SKILL_LEVEL=1
	) AS HIGH,
    (SELECT LAN.LANGUAGE FROM USERLANG ULAN
		INNER JOIN LANGUAGE LAN
		ON ULAN.LANGUAGE_ID = LAN.LANGUAGE_ID
		WHERE ULAN.STUDENT_ID = ULAN_MAIN.STUDENT_ID
		AND ULAN.SKILL_LEVEL=2
    ) AS MEDIUM
FROM USERLANG ULAN_MAIN
ORDER BY STUDENT_ID ASC;
```

```QUERY3-2
SELECT 
	(SELECT LAN.LANGUAGE FROM USERLANG ULAN
		INNER JOIN LANGUAGE LAN
		ON ULAN.LANGUAGE_ID = LAN.LANGUAGE_ID
		WHERE ULAN.STUDENT_ID = ULAN_MAIN.STUDENT_ID
		AND ULAN.SKILL_LEVEL=1
	) AS HIGH,
	count((SELECT LAN.LANGUAGE FROM USERLANG ULAN
		INNER JOIN LANGUAGE LAN
		ON ULAN.LANGUAGE_ID = LAN.LANGUAGE_ID
		WHERE ULAN.STUDENT_ID = ULAN_MAIN.STUDENT_ID
		AND ULAN.SKILL_LEVEL=2
    ))AS MEDIUM
FROM USERLANG ULAN_MAIN
GROUP BY HIGH
ORDER BY HIGH ASC;
```
